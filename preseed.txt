Bios Disable Other Boot Options to Force Boot from USB
Bios Setting > Advanced > CSM Configuration > Other PCI devices [Legacy]

### 1. Localization
d-i debian-installer/locale string en_ZA.UTF-8
d-i localechooser/languagelist select en
d-i localechooser/continentlist select Africa
d-i localechooser/countrylist/Africa select ZA
d-i debian-installer/country string ZA
d-i keyboard-configuration/xkb-keymap select us
d-i keyboard-configuration/layoutcode string us

### 2. Critical Installation Control
# Set priority to critical to suppress non-essential questions.
d-i debconf/priority string critical
# Network halting directives removed. Install will continue automatically after DHCP.

### 3. Network configuration
d-i netcfg/get_hostname string gppos
d-i netcfg/get_domain string gppos

d-i netcfg/domain string gppos

### 7. Use Global Debian CDN Mirror

### 8. Apt setup
d-i apt-setup/services-select multiselect updates
d-i apt-setup/non-free boolean true
d-i apt-setup/contrib boolean false

### 9. Package selection
popularity-contest popularity-contest/participate boolean false
tasksel tasksel/first multiselect standard system utilities
tasksel tasksel/desktop multiselect none
d-i pkgsel/include string wget git sudo lshw gnome-core net-tools linux-perf zip

### 10. Boot loader installation
d-i grub-installer/only_debian boolean true
d-i grub-installer/with_other_os boolean false
# FIX: Explicitly set the boot device to the primary disk to prevent prompting.
d-i grub-installer/bootdev string /dev/sda

### 11. Initialize Final System
d-i preseed/late_command string in-target wget -P /opt/ https://git.gppos.net/pos/debian/-/raw/master/setup.sh; in-target chmod +x /opt/setup.sh; in-target /opt/setup.sh
